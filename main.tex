\begin{luacode*}
function mymult(n)
  for j = 1, 10, 1 do
    for i = n-5+1, n do
      tex.print(string.format('%d & $\\times$ & %d & = & %d ', i, j, (i * j)))
      if i < n then
        tex.print(' & ')
      else
        tex.print("\\\\\\hline")
      end
    end
  end
end
\end{luacode*}
